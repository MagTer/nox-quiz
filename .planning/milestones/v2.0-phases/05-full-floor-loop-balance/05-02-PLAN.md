    Method: enterCombat():
      DungeonState.advanceRoom();
      const ds = DungeonState.get();
      const floorDef = FloorConfig.getFloor(_currentFloor);
      const isBoss = ds.room >= CONFIG.DUNGEON.ROOMS_PER_FLOOR;
      const enemyHP = isBoss
        ? Math.round(floorDef.hp * CONFIG.DUNGEON.BOSS_HP_MULT)
        : floorDef.hp;
      // CombatEngine.startCombat calls DungeonState.init internally — this would reset HP.
      // For rooms 2-5 on the same floor, we must NOT call startCombat (which calls init).
      // Instead, seed enemy HP directly and transition FSM.
      // Only call startCombat for the very first room of each floor (room === 1 at this point).
      // For subsequent rooms: FSM must be in EXPLORE (transitioned there after kill/loot).
      if (GameFSM.getState() !== 'EXPLORE') GameFSM.reset();
      // Set floorDef on CombatEngine by calling startCombat — but startCombat calls DungeonState.init
      // which resets playerHP. We need to preserve playerHP between rooms.
      // Solution: call startCombat only to set floorDef + FSM, then immediately restore playerHP.
      const savedHP = DungeonState.get().playerHP;
      // startCombat transitions FSM to COMBAT and sets floorDef on CombatEngine
      // It also calls DungeonState.init (resetting HP) — we fix that below.
      // BUT: for rooms 2+ on the same floor, init resets loot too. This is wrong.
      // Correct approach: do NOT call CombatEngine.startCombat for room-to-room transitions.
      // Instead, expose a lighter method OR work around it.
      // Since CombatEngine has no setFloor() method, we must call startCombat and patch HP+loot after.
      // This is acceptable for Phase 5 scope (Phase 6 can refactor CombatEngine).
      const savedLoot = DungeonState.get().loot;
      CombatEngine.startCombat(_currentFloor);          // resets HP + loot via DungeonState.init
      DungeonState.setPlayerHP(savedHP);                // restore HP from before this room
      // Restore loot: re-apply saved items
      if (savedLoot.sword)   DungeonState.applyLoot('sword');
      if (savedLoot.shield)  DungeonState.applyLoot('shield');
      // potions are consumed on pickup; not re-applied
      DungeonState.setEnemyHP(enemyHP);                 // override with correct HP (boss or normal)
      // Render combat panel with correct state
      const cs = CombatEngine.getState();
      DungeonRenderer.renderCombat({
        floor:    _currentFloor,
        room:     ds.room,
        enemy:    floorDef.enemy,
        emoji:    floorDef.emoji,
        enemyHP:  enemyHP,
        playerHP: savedHP
      });
      App.transition('combat');
      CombatInputHandler.beginCombat();

    Method: afterKill():
      _floorEnemiesDefeated += 1;
      const ds = DungeonState.get();
      // XP was already awarded by CombatEngine.resolveAnswer (via PlayerState.addXp)
      // Track floor XP for summary — read current PlayerState XP delta is hard;
      // simpler: accumulate floorDef.xpReward per kill.
      const floorDef = FloorConfig.getFloor(_currentFloor);
      const isBoss = ds.room >= CONFIG.DUNGEON.ROOMS_PER_FLOOR;
      const xpThisKill = isBoss
        ? (FloorConfig.getFloor(_currentFloor).xpReward * CONFIG.DUNGEON.BOSS_HP_MULT)
        : floorDef.xpReward;
      _floorXpEarned += xpThisKill;
      // Determine loot drop
      const shouldDrop = isBoss
        ? true
        : Math.random() < CONFIG.DUNGEON.LOOT_DROP_RATE;
      if (shouldDrop) {
        const items = ['sword', 'shield', 'potion'];
        const itemName = items[Math.floor(Math.random() * items.length)];
        const LOOT_DISPLAY = {
          sword:  { emoji: '🗡️', name: 'Sword of Accuracy' },
          shield: { emoji: '🛡️', name: 'Shield of Resilience' },
          potion: { emoji: '🧪', name: 'Health Potion' }
        };
        DungeonState.applyLoot(itemName);
        // Potion: restore HP immediately on pickup
        if (itemName === 'potion') {
          const healedHP = Math.min(
            CONFIG.DUNGEON.PLAYER_HP,
            DungeonState.get().playerHP + CONFIG.DUNGEON.POTION_HEAL
          );
          DungeonState.setPlayerHP(healedHP);
        }
        DungeonRenderer.renderLoot(LOOT_DISPLAY[itemName]);
        App.transition('loot');
      } else {
        // No loot — go directly to next room or floor summary
        DungeonRunner.afterLoot();
      }

    Method: afterLoot():
      const ds = DungeonState.get();
      if (ds.room >= CONFIG.DUNGEON.ROOMS_PER_FLOOR) {
        // Floor cleared — show summary
        DungeonRenderer.renderFloorComplete({
          floor:            _currentFloor,
          enemiesDefeated:  _floorEnemiesDefeated,
          xpEarned:         _floorXpEarned,
          hpRemaining:      ds.playerHP
        });
        // Special headline for Final Boss floor
        if (_currentFloor >= CONFIG.DUNGEON.MAX_FLOORS) {
          document.getElementById('floor-summary-headline').textContent = 'Dungeon Cleared!';
        }
        App.transition('floor-summary');
      } else {
        // More rooms remain — start next combat
        DungeonRunner.enterCombat();
      }

    Method: advanceFloor():
      if (_currentFloor < CONFIG.DUNGEON.MAX_FLOORS) {
        _currentFloor += 1;
        _floorEnemiesDefeated = 0;
        _floorXpEarned = 0;
        DungeonState.init(_currentFloor);
        DungeonRunner.updateDungeonMap(_currentFloor);
        App.transition('dungeon-map');
      } else {
        // Dungeon complete — return to quiz
        CombatEngine.reset();
        GameFSM.reset();
        App.transition('quiz');
      }

    Method: retryFloor():
      _floorEnemiesDefeated = 0;
      _floorXpEarned = 0;
      // ADHD-03: DungeonState.init resets HP/loot only — PlayerState XP untouched
      DungeonState.init(_currentFloor);
      GameFSM.reset();
      CombatEngine.reset();
      DungeonRunner.updateDungeonMap(_currentFloor);
      App.transition('dungeon-map');

    All method bodies use textContent exclusively (no innerHTML for content). No magic numbers — all references use CONFIG.DUNGEON.* constants.

    IMPORTANT — CombatEngine.resolveAnswer() already transitions FSM to LOOT (on kill) or DEAD (on death). After those transitions, DungeonRunner.afterKill() is called from CombatInputHandler. App.transition('loot') inside afterKill() writes data-screen but does NOT call GameFSM.transition (App.transition is a screen routing call, not FSM). This means FSM remains in LOOT state. enterCombat() calls GameFSM.reset() before startCombat() to return to EXPLORE — so startCombat can proceed.

    Similarly after death: FSM is in DEAD. retryFloor() calls GameFSM.reset() → FSM back to EXPLORE. startCombat can be called again.
