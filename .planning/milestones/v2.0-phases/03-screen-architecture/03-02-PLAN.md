    This task requires careful sequencing because App.nextQuestion() calls Renderer.showQuestion() and QuestionSelector.selectNext(), which are defined inside DOMContentLoaded. The solution is a two-part pattern:

    PART A — Define App OUTSIDE DOMContentLoaded (before the document.addEventListener line):

    Remove the existing App object from inside DOMContentLoaded. In its place, define App outside DOMContentLoaded with this structure:

      const App = (() => {
        let mode = 'quiz';  // 'quiz' | 'dungeon'

        // renderScreen is the only function that sets data-screen on main.
        // Called exclusively from transition() — no other code path may call it.
        function renderScreen(name) {
          const main = document.getElementById('game-board');
          if (main) main.setAttribute('data-screen', name);
        }

        return {
          get mode() { return mode; },

          transition(screenName) {
            mode = (screenName === 'quiz') ? 'quiz' : 'dungeon';
            renderScreen(screenName);
          },

          // nextQuestion is a stub here; DOMContentLoaded wires the real implementation
          // via App._nextQuestion after Renderer and QuestionSelector are available.
          nextQuestion() {
            if (typeof App._nextQuestion === 'function') App._nextQuestion();
          }
        };
      })();
      window.App = App;

    PART B — Inside DOMContentLoaded, after Renderer and QuestionSelector are available (where the old App object was), wire the implementation stub:

      // Wire App.nextQuestion() to the real implementation now that Renderer is available
      App._nextQuestion = function() {
        const q = QuestionSelector.selectNext(PlayerState);
        InputHandler.currentQuestion = q;
        Renderer.showQuestion(q);
      };

    This preserves the exact call sites in InputHandler.handleAnswer (the setTimeout callback calls App.nextQuestion()) and in the bootstrap section (App.nextQuestion()) without requiring any changes at those call sites.

    Do NOT rename any other module. Do NOT change the order of the GameFSM, FloorConfig, DungeonState, or CombatEngine definitions above DOMContentLoaded.

    The window.App = App line must appear immediately after the App IIFE closing parenthesis, before document.addEventListener('DOMContentLoaded').

    IMPORTANT: Do NOT add DOM.main to the DOM cache in this task — that belongs in Task 2. The renderScreen function in this task uses document.getElementById('game-board') directly because DOMContentLoaded may not have run yet when the function is defined (it is only called after DOMContentLoaded in practice, but getElementById is safe to call at any time).
  </action>
